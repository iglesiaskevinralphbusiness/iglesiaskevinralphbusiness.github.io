Angular

Installation

Step 1: install nodejs

Step 2: npm install -g @angular/cli

Step 3: ng new my-app --style=scss --routing

Step 4: cd my-app

To run the app: ng serve

To run the app with port: ng serve --port 4205

Sub Routing / Module Routing

import { StudentComponent } from './student/student.component';
import { StudentListComponent } from './student-list/student-list.component';
import { StudentDetailComponent } from './student-detail/student-detail.component';

const routes: Routes = [
    {
        path: 'student',
        component: StudentComponent,
        children: [
            {
                path: 'list',
                component: StudentListComponent
            },
            {
                path: 'detail',
                component: StudentDetailComponent
            }
        ]
    }
];

Without Parameter

IN TS - Request (data.service.ts)

...
import { HttpClient } from '@angular/common/http';

constructor(private http: HttpClient) {}

getUsers() {
    return this.http.get('https://jsonplaceholder.typicode.com/users');
}

IN TS - Implementation (users.components.ts)

...
import { DataService } from '../data.service';
import { Observable } from 'rxjs';
...
export class UsersComponent implements OnInit {
	
	users$: Object;

	constructor(private data: DataService) { }

	ngOnInit() {
		this.data.getUsers().subscribe(
			data => {
				this.users$ = data;
				console.log(data);
			},
			error => console.log(error)
		)
	}
}

IN HTML (user.components.html)

<ul>
	<li *ngFor="let user of users$">
		<a routerLink="/details/{{user.id}}">{{ user.name }}
	</li>
</ul>

With Parameter

IN TS - Request (data.service.ts)

...
import { HttpClient } from '@angular/common/http';

constructor(private http: HttpClient) {}

getUser(userId) {
    return this.http.get('https://jsonplaceholder.typicode.com/users/' + userId);
}

IN TS - Implementation (details.components.ts)

...
	import { DataService } from '../data.service';
	import { Observable } from 'rxjs';
	import { ActivatedRoute } from '@angular/router';
	...
	export class DetailsComponent implements OnInit {
		
		user$: Object;
	
		constructor(private data: DataService, private route: ActivatedRoute) { 
			this.route.params.subscribe(params => this.user$ = params.id)        //---> get the url parameters ex. https://jsonplaceholder.typicode.com/users/post/123
		}
	
		ngOnInit() {
			this.data.getUser(this.user$).subscribe(
				data => this.user$ = data
			)
		  }
	}
