import React, {useState } from 'react';
		
const Counter = () => {
	const [counter, setCounter] = useState(0); //counter remaind value as 0, and setCounter is the variable that can be modified.

		return (
			<Button title="+" onPress={() => {

			<Button>
			<Text>{{ setCounter }}</Text>
		);
	}
}

import React, { Component } from 'react'
import { Text, View, FlatList } from 'react-native'

export default class App extends React.Component {
	
	const friends = [
		{ name: "Kevin", key: "1" },
		{ name: "Ralph", key: "2" },
		{ name: "Emote", key: "3" }
	]

	render() {
		return (
			<FlatList
				horizontal //remove this you want to show the list in  vertical
				showHorizontalScrollIndicator={false} //set this if you want to hide scroll in horizontal
				data={friends}
				renderItem={ ({item}) => {
					return {item.name}
				}}
				keyExtractor={friend => friend.name}
		);
	}
}

Reducer

Assigning state with function

import React, { useReducer } from 'react';

const reducerFunction = (action, value) => {
	switch(action){
		case 'red':
			return {...state, red: state.red + value};
		case 'green':
			return {...state, green: state.green + value};
		default
			return state;
	}
}

const SampleReducer = () => {
	const [state, dispatch] = useReducer(reducerFunction, { red: 0, green: 0});

	return (
		<View>
			<Button 
				onPress = {() => dispatch('red', 10)} //increase value 10 every click
				title = "Increase Red"
			/>
			<Button 
				onPress = {() => dispatch('green', -10)} //decrease value 10 every click
				title = "Increase Green"
			/>
		</View>
	)
}

import React, { Component } from 'react'
import { View, Text, TouchableOpacity, TextInput, StyleSheet } from 'react-native'

class Inputs extends Component {
	state = {
		email: '',
		password: ''
	}
	handleEmail = (text) => {
		this.setState({ email: text })
	}
	render() {
		return (
			<View>
			<TextInput
				underlineColorAndroid = "transparent"
				placeholder = "Email"
				placeholderTextColor = "#9a73ef"
				autoCapitalize = "none"
				autoCorrect={false}
				onChangeText = {this.handleEmail}
				value = "kevin"
			/>
		<View>
		)
	}
}
export default Inputs
