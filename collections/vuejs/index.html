Properties

Computed Properties Vs Methods Properties

If you declared a rendering function in the page:

Computed will execute the function only if value of related data variable whithin that function is changed.

Method will always run every time data variable are changed. (Disadvantage if your functions code is too long script which is executed but not related on changes, it only interupt the speed performance of app)

Calling function:

Computed called without parenthesis

Method called with parenthesis

<div id="app">
		<button v-on:click="counter1++">Increase</button>
		<button v-on:click="counter1--">Decrease</button>
		<button v-on:click="counter2++">Increase Counter 2</button>
		<p>Counter 1: {{ counter1 }}</p>
		<p>Counter 2: {{ counter2 }}</p>
		<p>Result: {{ renderFlag }} | {{ renderCounter() }}</p>
	</div>
new Vue({
	el: '#app',
	data: {
			counter1: 0,
			counter2: 0,
			flag: ''
	},
	computed: {
			renderFlag: function(){
				console.log('Computed Triggered'); // will not triggere if you click the Increase Counter 2 button
				return this.counter1 > 5 ? 'Greater than 5' : 'Less than 5';
			}
	},
	methods: {
			renderCounter: function(){
				console.log('Method Triggered');
				return this.counter1;
			}
	}
});

Status.vue (create a eventBus listener)

<template>
	<div>
		<p>Child Status: {{ status }}</p>
	</div>
</template>

<script>
	import { eventBus } from './main.js';

	export default {
		props: {
			status: {
				type: String
			},
		},
		created(){
			eventBus.$on('handleBusStatus', (status) => {
				this.status = status;
			});
		}
	}
</script>

StatusEdit.vue (access the eventBus listener)

<template>
	<div>
		<p>Child Status: {{ status }}</p>
		<button @click="editStatus">Edit</button>
	</div>
</template>

<script>
	import { eventBus } from './main.js';

	export default {
		props: {
			status: {
				type: String
			},
		},
		methods: {
			editStatus(){
				this.status = 'Bus emitted';
				eventBus.$emit('handleBusStatus', this.status);
			}
		}
	}
</script>

Global Custom Directive

Main.js - declare the directive to make it global

import Vue from 'vue'
	import App from './App.vue'
	import Home from './Home.vue'
	
	Vue.component('Home', Home);
	
	Vue.directive('highlight', {
		bind(el, binding, vnode){
		//el.style.backgroundColor = 'green'
		//el.style.backgroundColor = binding.value
		var delay = 0;
		if(binding.modifiers['delayed']){
			delay  =  2000;
		}
		setTimeout(() => {
			if(binding.arg == 'background'){
			el.style.backgroundColor = binding.value;
			} else {
			el.style.color = binding.value;
			}
		}, delay)
		}
	})
	
	export const eventBus = new Vue();
	
	new Vue({
		el: '#app',
		render: h => h(App)
	})

Local Mixins

Create a new copy of vue instance and merge to another vue instance


Main.js

<template>
	<div>
		<p>{{ text }}</p>
		<ul>
			<li v-for="fruit in fruits">{{ fruit }}</li>
		</ul>
	</div>
</template>

<script>
	import { fruitMixins }  from './fruitMixins';

	export default {
		mixins: [fruitMixins],
		data(){
			return {
				text: 'Hello World'
			}
		},
	}
</script>

fruitMixins.js

export const fruitMixins = {

		}
	}
}

				fname: '',
				lname: ''
			}
		}
	},

		submit(){
			this.$http.post('/account', this.account)

					console.log(error);
				});
		}
	}

		return {
			list: []
		}

			this.$http.get('/account')

					this.list = response.body;

					console.log(error);
				});
		}
	}
}

Interceptors

Intercepting Request - is appliend before the request will be made.

Intercepting Response - is appliend after the request. You can change the response value or make an if statement on it.


Main.js

Vue.http.interceptors.push((request, next) => {
	//intercepting request
	//example change method request of PUT if method is POST
	if(request.method == 'POST'){
		request.metho = 'PUT';
	}

	//intercepting response
	next(response => {
		console.log(response); //add your code here
	});
});

Custom Resouce (Shorten the request)

    export default {
	data(){
		return {
			account: {
				fname: '',
				lname: '',
			},
			resource: {} //call an resource variable
		}
	},
	created() {
			//create all actions and url api
		const actions = {
			saveAccount: { method: 'POST', url: '/account' },
			getAccount: { method: 'GET' },
		}
		//set the variable as resource
		this.resource = this.$resource('{node}', {}, actions);
	},
	method: {
		//use get and post resource
		saveAccount(){
			this.resource.saveAccount(this.account); //to use a request
		},
		getAccount(){
			this.resource.getAccount({node: '/account'})
				.then(response => {
					console.log(response);
				}, error => {
					console.log(error);
				})
		}
	}
}

Passing Parameters to Router Link

Basic Way

If you want to use dynamic <router-link>, use v-bind:to or :to in tag to use variables

<router-link tag="button" :to="'/user' + $route.params.id + '/edit'">Edit</router-link>

Better Way / Advance

Routes.js (add a name to your router)

{ path: ':id/edit', component: UserEdit, name: 'userEdit' },

To use inside Html

<router-link tag="button" :to="{ name: 'userEdit', params: { id: $route.params.id } }">Edit</router-link>

To use inside Script

this.$route.push({ name: 'userEdit', params: { id: $route.params.id })

Using Query Parameters

Getters

Performs any calculations and then will access from different component. This is usefull for global task that need to work with multiple components.


/store/store.js (Create getters)

import Vue from 'vue';
import Vuex from 'vuex';

Vue.use(Vuex);

export const store = new Vuex.Store({
	state: {
		counter: 0,
	},
	getters: {  //create getters here
		doubleCounter: state => {
			return state.counter * 2;
		},
        stringCounter: state => {
            return state.counter + ' Clicks';
        }
	}
})

Mutations

Centralize or common updating state. This is usefull for global task that need to work with multiple components.


/store/store.js

import Vue from 'vue';
import Vuex from 'vuex';

Vue.use(Vuex);

export const store = new Vuex.Store({
	state: {
		counter: 0,
	},
	mutations: {   //create mutations here
		increment: (state, payload) => {
			state.counter = state.counter + payload; //payload is a parameters you passed
		},
		decrement: state => {
			state.counter--;
		}
	}
})

Basic Updating

Counter.vue

<script>
	export default {
		methods: {
			increment() {
				this.$store.commit('increment', 100); //use this //100 is passing payload value
			},
			decrement() {
				this.$store.commit('decrement'); //use this
			}
		}
	}
</script>
